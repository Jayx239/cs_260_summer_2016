\begin{algorithm}
	\KwData{One Integer Array as input}
	\KwResult{Computed Larges Value in Array}
int GetLargestValue(int[] A)\\
\{\\
	int largest = A[0]\;
	for(int i=0; i<A.size(); i++)\\
	\{\\
		if (A[i] > largest)
			largest = A[i]\;
	\{\\
	
	return largest\;
\}	
\end{algorithm}
